                          系统同步

                          sync # 将数据由内存同步到硬盘中

                          查看帮助

                          man sync # 查看sync的帮助信息

                          sync --help # 同上

                          关机

                          poweroff # 立即关机并关闭电源 --- > 不通电

                          halt 立即关机但是不关闭电源 ---> 通电

                          shutdown -h now # 立即关机

                          shutdown -h 10 # 十分钟后关机

                          shutdown -h 08:40 # 今天八点四十关机

                          shutdown -c # 取消关机

                          重启

                          reboot # 立即重启

                          shutdown -r now # 立即重启

                          shutdown -r 10 # 十分钟后重启

                          shutdown -c # 取消重启

                          # 目录操作

                          查看路径

                          pwd	# 查看当前所在文件的路径

                          切换路径

                          cd /			# 进入根目录
cd ~			# 进入当前用户的主目录
cd ..			# 进入父级目录
cd usr			# 进入当前目录下的usr目录
cd /usr/bin      # 进入 /usr/bin 目录

                          查看子目录及文件列表

                          ls			# 以列表形式查看
ls -a		# 查看所有文件，包含隐藏文件
ls -l		# 以详细信息形式查看
ls -al		# 以详细信息形式查看所有文件

                          详细列表说明

                          ls -l	# 以详细信息形式查看所有文件

                          列 位置 说明
                          第一列 第1个字母 文件类型：d=目录 -=文件 l=链接 b=外部接口 c=串行接口设备（鼠标、键盘等） p=管理文件 s=套接字文件
                          第一列 第2、3、4个字母 所有者的权限：r=可读 w=可写 x=可执行 -=无权限
                          第一列 第5、6、7个字母 与所有者同组用户的权限：r=可读 w=可写 x=可执行 -=无权限
                          第一列 第8、9、10个字母 其他用户的权限：r=可读 w=可写 x=可执行 -=无权限
                          第二列 直接子目录个数，需要减去2才是真实数量
                          第三列 所有者名称
                          第四列 所有者所在组名称
                          第五列 文件大小
                          第六列 文件修改时间，格式 月 日 时:分
                          第七列 文件或目录名称

                          创建目录

                          mkdir young		    			       # 在当前目录下创建young目录
mkdir /young/test		    		   # 在指定目录下创建test目录
mkdir -p /young/test1/test2/test3		# 递归创建 test1/test2/test3 目录

                          删除目录

                          rmdir young							  # 删除young目录，young是一个空目录  如果young不为空，会提示 rmdir: failed to remove `young`: Directory not empty
rmdir -p /young/test1/test2/test3		# 递归删除 /young/test1/test2/test3 目录
rm -rf young						  # 强制删除非空目录
rm -rf *							  # 强制删除该目录下的子目录及文件

                          目录拷贝

                          cp -r <原目录或文件> <新目录或文件>
cp -r test1 test2	# 拷贝test1到test2
cp -r test1 test2/test	# 拷贝test1到test2并取一个别名test

                          移动目录

                          mv <原目录或文件> <新目录或文件>
mv test2/naihe .	# 将test2下的naihe移动到当前目录下

                          重命名目录

                          mv test2 test	# test不能与已有目录或文件重名 否则就是移动到该目录名下

                          搜索子目录或文件

                          find naihe	# 在当前目录下搜索 naihe 目录或文件
find test*	# 在当前目录下搜索以 test 开头的目录或文件，* 为通配符
find -type f -name test		# 在当前目录下搜索 test 文件
find -type d -name test		# 在当前目录下搜索 test 目录

                          # 全系统搜索 passwd 文件，并显示位置
# 无法搜索普通文件，只能搜索系统文件及命令

whereis passwd

                          # 文件操作

                          创建文件

                          touch test.txt	# 在当前目录下创建一个 test.txt 文件

                          创建链接

                          ln test.txt t1.txt		# 为 text.txt 文件创建硬链接 t1.txt
ln -s test.txt t2.txt	# 为 text.txt 文件创建软链接 t2.txt

                          硬链接和软链接说明

                          **硬链接：**是指多个链接指向同一个文件，删除一个链接不影响另一个链接使用

                          **软链接：**是指为文件创建一个快捷方式，删除了源文件软链接也无法使用

                          # 压缩

                          ==不能压缩目录==

                          gzip 方式

                          # 压缩目录下所有文件
# 压缩后生成 *.gz文件 并删除源文件
gzip *

# 压缩 t1.txt 文件
gzip t1.txt

                          bzip2方式

                          # 使用 bzip2 压缩 先安装 bzip2支持
yum -y install bzip2
# 压缩 t3 文件
# 压缩成功后，删除源文件，并生成 *.bz2 文件
bzip2 t3
# 压缩所有文件
bzip2 *

                          # 解压

                          gzip

                          gzip -dv *
# 解压所有以 *.gz 形式的压缩文件   --->    t.gz

                          bzip2

                          bzip2 -d *
# 解压所有以 *.bz2 形式的压缩文件

                          gunzip

                          gunzip *
# 解压目录下所有以 *.gz 形式的压缩文件
# 和 gzip -dv 差不多
# 不用加 -dv

                          # 查看压缩文件信息

                          # 查看目录下所有 gz 压缩文件信息
# 以列表形式显示
gzip -l *



# compressed=压缩后大小 uncompressed=压缩前大小 ratio=压缩比率 uncompressed_name=压缩前的文件名

                          # 打包与解包

                          打包

                          ==可以打包目录==

                          # 将 t1 文件 打包为 t1.tar 文件
# 不删除源文件
tar -c -f t1.tar t1

                          # 将 t2 文件打包为 t2.tar 文件
# 显示打包的文件名
tar -cv -f t2.tar t2

                          # 将 t3 文件打包为 t3.tar.gz 文件
tar -zcvf t3.tar.gz t3

# 将 t4 文件打包为 t4.tar.bz 文件
tar -jcvf t4.tar.bz t4

                          # tar 可以将文件打包为 *.gz *.bz 文件
tar -c -f t1.gz t1
tar -c -f t1.bz t1

                          解包

                          # 将文件解压到本目录
tar -xf t1.tar

# 将 t1.tar 解包到 /home 目录下，并命名为 t 文件
tar -xf /home/t t1.tar

# 将 t1.tar 解包到本地目录，并显示解包过程
tar -xvf t1.tar

                          # 将 t3.tar.gz 解包到本地目录
tar -zxvf t3.tar.gz

                          # 将 t1.tar.bz 解包到本地目录
tar -jxvf t1.tar.bz

                          # 网络管理

                          查看网络

                          ip addr

                          重启网络服务

                          service network restart

                          systemctl restart network

                          启用网络连接

                          • 进入网络配置文件

                            cd /etc/sysconfig/network-scripts

                          • 编辑网络配置文件

                            vi ifcfg-ens32

                          • 启用网络

                            ONBOOT=yes

                          • 重启网络

                            service network restart

                          • 查看网络

                            ip addr

                          固定IP地址配置

                          • 打开指定网卡配置文件

                            vi /etc/sysconfig/network-scripts/ifcfg-ens33

                          • 查看当前网络详情

                            nmcli device show

                          • 修改配置

                            # 启用静态IP
# static=固定IP dhcp=动态获取IP
BOOTPROTO=static

                            # 指定IP地址
IPADDR=192.168.198.137
NETMASK=255.255.255.0
GATEWAY=192.168.198.2
DNS1=192.168.198.2
DNS2=192.168.198.2

                          • 重启网络

                            service network restart

                          • 查看网络设备信息

                            # 查看网络设备基本信息
nmcli

                            # 查看网络设备完整信息
nmcli device show

                            # 查看活动链接概述
nmcli connection show

                          # 网络访问

                          访问网络

                          curl www.baidu.com
# 访问百度

                          保存网页

                          curl www.baidu.com >> baidu.html

                          curl -O baidu2.html www.baidu.com

                          下载文件

                          curl -O http://s1.bdstatic.com/r/www/cache/bdorz/baidu.min.css
# 下载百度的 css 文件

                          curl -# -O www.baidu.com/img/bd_logo1.png
# 下载图片并显示进度条

                          curl -c cookie www.baidu.com
# 保存 cookie 信息到 cookie 文件

                          curl -D response www.baidu.com
# 保存响应头信息到 response 文件

                          FTP 上传与下载

                          curl -O ftp://username:password@www.ftp.com/image.jpg
# 从 www.ftp.com FTP服务器中下载 image.jpg 图片

                          curl -T image.jpg -u username:password ftp://www.ftp.com
# 将图片 image.jpg 上传至 www.ftp.com FTP服务器

                          # VI编辑器

                          模式说明

                          **命令模式：**使用 vi/vim 打开文件后，会进入命令模式

                          **编辑模式：**在命令模式下输入 i/a 进入编辑模式

                          **末行模式：**在命令模式下输入 ： 进入末行模式 然后输入 esc 即可退出

                          w 保存
q 退出
wq 保存退出
ZZ 如果文档有改动则保存退出，否则直接退出
w 文件名	另存为
r 文件名	读取另一篇文章数据，追加到光标所在行后面
n1,n2 w 文件名	 将n1行到n2行的数据保存为新文件
set nu 显示行号
set nonu 取消行号

                          # YUM 管理

                          # 显示所有可安装的软件
yum list
# 显示软件名为java开头的软件
yum list java*
# 过滤软件名包含 jdk 的软件
yum list | grep jdk
# 查看可更新的软件包
yum list updates
# 查看已安装的软件包
yum list installed
# 搜索软件名称包含 jdk 的软件
yum search jdk

                          软件安装

                          yum install -y 软件名

                          软件更新

                          # 显示可更新软件包
yum check-update
# 更新所有软件包
yum update -y
# 更新某一个软件
yum update -y 软件名

                          软件卸载

                          yum remove -y 软件名

                          清除缓存

                          yum clean all

                          构建缓存

                          # 把远程服务器的包信息缓存至本地
yum makecache

                          更新 YUM 源

                          • 备份 YUM 配置文件

                            mv /etc/yum.repos.d/CentOS-Base.repo /etc/yum.repos.d/CentOS-Base.repo_bak

                          • 下载阿里源

                            curl -O http://mirrors.aliyun.com/repo/Centos-7.repo

                          • 配置 YUM 源

                            mv Centos-7.repo /etc/yum.repos.d/CentOS-Base.repo

                          • 重建缓存

                            yum makecache
